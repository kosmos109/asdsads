#-*- coding: cp1251 -*-

info = {
    'name': None,
    'age': None,
    'gender': None
}

actions = [
    'Поговорить с незнакомцем',
    'Исследовать окрестности',
    'Найти способ выбраться',
    'Попытаться вызвать помощь',
    'Найти ответы'
]

inventory = []

visited_places = set()

def start_game():
    print("Добро пожаловать в нашу игру!")
    info['name'] = input("Как вас зовут? ")
    info['age'] = input("Сколько вам лет? ")
    info['gender'] = input("Какого вы пола? ")

    print(f"Приятно познакомиться, {info['name']}!")
    print("Вы очнулись в неизвестном месте...")

def choose_action():
    print("Выберите действие:")
    for i, action in enumerate(actions):
        print(f"{i+1}. {action}")

    choice = int(input("Ваш выбор: "))

    if choice == 1:
        talk_with_stranger()
    elif choice == 2:
        explore_surroundings()
    elif choice == 3:
        find_a_way_out()
    elif choice == 4:
        call_for_help()
    elif choice == 5:
        find_answers()
    else:
        print("Некорректный выбор. Попробуйте снова.")
        choose_action()

def talk_with_stranger():
    print("Вы подошли к незнакомцу и начали с ним разговор.")
    print("Незнакомец рассказал вам одну важную информацию.")

    info['important_info'] = "Важная информация от незнакомца."

    choose_action()

def explore_surroundings():
    print("Вы исследуете окрестности и находите некоторые предметы.")
    print("Добавляем предметы в инвентарь!")

    inventory.append("Предмет 1")
    inventory.append("Предмет 2")

    choose_action()

def find_a_way_out():
    if "Предмет 1" in inventory and "Предмет 2" in inventory:
        print("Вы успешно нашли способ выбраться!")
        print("Поздравляю, вы победили!")
    else:
        print("Вы еще не нашли все необходимые предметы.")
        choose_action()

def call_for_help():
    if 'important_info' in info and 'Предмет 1' in inventory:
        print("Вы вызвали помощь и передали важную информацию и предмет.")
        print("Помощь пришла!")
        print("Поздравляю, вы победили!")
    else:
        print("Вы не можете вызвать помощь.")
        choose_action()

def find_answers():
    if 'important_info' in info:
        print("Вы нашли ответы на свои вопросы!")
        print("Поздравляю, вы победили!")
    else:
        print("Вы еще не знаете достаточно информации.")
        choose_action()

start_game()
choose_action()
